    <script>

        function generate() {

            const checks = document.querySelectorAll('input:checked');

            const ids = Array.from(checks).map(c => c.value).join(',');

            if(!ids) return alert("Choisis au moins une équipe !");

            

            const baseUrl = window.location.origin;

            const link = `${baseUrl}/api/cal?teams=${btoa(ids)}`;

            document.getElementById('url').innerText = link;

            document.getElementById('result').classList.remove('hidden');

        }
